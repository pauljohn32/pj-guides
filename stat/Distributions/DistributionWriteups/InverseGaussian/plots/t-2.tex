\begin{Sinput}
 par(mfrow=c(4,3))
 for (i in 1:4){
      for (j in 1:3) {
          mu <- 3*i
          lambda <- 20*j
          xrange <- seq(from = 0.0, to = 3*mu, by = 0.02)
          mainlabel<- paste("IG(",mu,",",lambda,")", sep = "")
          xprob <- dinvgauss(xrange, mu = mu, lambda = lambda)
          plot(xrange, xprob, type = "l", main = mainlabel, xlab = "possible values of x", ylab = "probability of x")
      }
  }
\end{Sinput}
