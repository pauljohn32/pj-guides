%% ">=stealth" sets the arrow head style
%% "semithick" sets the line width (0.6 pt)
\begin{tikzpicture}[>=stealth,semithick]

% observed variables
\node[ov] (y1) at (0,-1)      {$y_1$};
\node[ov] (y2) [below of=y1]  {$y_2$};
\node[ov] (y3) [below of=y2]  {$y_3$};
\node[ov] (y4) [below of=y3]  {$y_4$};
\node[ov] (y5) [below of=y4]  {$y_5$};
\node[ov] (y6) [below of=y5]  {$y_6$};

% latent variables
\node[lv] (f1) at (3, -2)  {$\eta_1$};
\node[lv] (f2) at (3, -5)  {$\eta_2$};

% paths (latent to observed)
\path[->] (f1) edge node[above,scale=0.6] {} (y1)
          (f1) edge node[above,scale=0.6] {} (y2)
          (f1) edge node[above,scale=0.6] {} (y3)
          (f1) edge node[above,scale=0.6] {} (y4)
          (f1) edge node[above,scale=0.6] {} (y5)
          (f1) edge node[above,scale=0.6] {} (y6)
          (f2) edge node[above,scale=0.6] {} (y1)
          (f2) edge node[above,scale=0.6] {} (y2)
          (f2) edge node[above,scale=0.6] {} (y3)
          (f2) edge node[above,scale=0.6] {} (y4)
          (f2) edge node[above,scale=0.6] {} (y5)
          (f2) edge node[above,scale=0.6] {} (y6);

% path (between latent vars)
\path[<->] (f1.east) edge [bend left=45] node[left,scale=0.8] {} (f2.east);

\end{tikzpicture}


\end{document}
